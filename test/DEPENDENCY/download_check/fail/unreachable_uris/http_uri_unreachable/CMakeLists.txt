## HTTP URI Unreachable Test
#
# Test DEPENDENCY failure when HTTP URI is unreachable
# TEST must FAIL
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.18)
    PROJECT(CMLIB_DEPENDENCY_HTTP_FAIL_TEST)
ENDIF()

LIST(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/../../../../../")
FIND_PACKAGE(CMLIB)

SET(UNREACHABLE_HTTP_URI "https://nonexistent.domain.invalid/file.zip")

CMLIB_DEPENDENCY(
    TYPE FILE
    URI "${UNREACHABLE_HTTP_URI}"
    OUTPUT_PATH_VAR output_path
)
