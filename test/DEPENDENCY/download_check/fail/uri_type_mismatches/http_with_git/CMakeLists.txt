## URI_TYPE GIT Mismatch Test
#
# Test DEPENDENCY failure when URI_TYPE GIT is specified but URI is not Git
# TEST must FAIL
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.18)
    PROJECT(CMLIB_DEPENDENCY_URI_TYPE_FAIL_TEST)
ENDIF()

LIST(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/../../../../../../")
FIND_PACKAGE(CMLIB)

SET(HTTP_URI "https://example.com/file.txt")

CMLIB_DEPENDENCY(
    TYPE FILE
    URI "${HTTP_URI}"
    URI_TYPE GIT
    OUTPUT_PATH_VAR output_path
)
