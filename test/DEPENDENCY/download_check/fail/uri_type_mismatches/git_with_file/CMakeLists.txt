## URI_TYPE FILE Mismatch Test (Git URI)
#
# Test DEPENDENCY failure when URI_TYPE FILE is specified but URI is Git
# TEST must FAIL
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.18)
    PROJECT(CMLIB_DEPENDENCY_URI_TYPE_FAIL_TEST)
ENDIF()

LIST(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/../../../../../../")
FIND_PACKAGE(CMLIB)
INCLUDE("${CMAKE_CURRENT_LIST_DIR}/../../../../../transform_git_uri.cmake")

TRANSFORM_GIT_URI_TO_HTTPS(
    URI "${CMLIB_REQUIRED_ENV_REMOTE_URL}/cmakelib-test.git"
    OUTPUT_VAR GIT_URI
)

CMLIB_DEPENDENCY(
    TYPE FILE
    URI "${GIT_URI}"
    URI_TYPE FILE
    GIT_PATH "TEST_MODULE.cmake"
    OUTPUT_PATH_VAR output_path
)
