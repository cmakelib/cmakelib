## URI_TYPE HTTP Mismatch Test
#
# Test DEPENDENCY failure when URI_TYPE HTTP is specified but URI is not HTTP
# TEST must FAIL
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.18)
    PROJECT(CMLIB_DEPENDENCY_URI_TYPE_FAIL_TEST)
ENDIF()

LIST(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/../../../../../../")
FIND_PACKAGE(CMLIB)

INCLUDE("${CMAKE_CURRENT_LIST_DIR}/../../../../../test_resources/test_resources.cmake")

TEST_RESROUCES_GET_FILE_URI("README.md" FILE_URI)

CMLIB_DEPENDENCY(
    TYPE FILE
    URI "${FILE_URI}"
    URI_TYPE HTTP
    OUTPUT_PATH_VAR output_path
)
